生成二元联合正态分布¶

$(X_1,X_2)\sim N\left(\mu,\Sigma\right)$$$X_{1}|X_{2}\sim N\left(\mu_{1}+\rho\frac{\sigma_{1}}{\sigma_{2}}\left(X_{2}-\mu_{2}\right),\sigma_{1}^{2}\left(1-\rho^{2}\right)\right)$$$$X_{1}=\mu_{1}-\rho\frac{\sigma_{1}}{\sigma_{2}}\mu_{2}+\rho\frac{\sigma_{1}}{\sigma_{2}}X_{2}+\epsilon$$$\epsilon\sim N\left(0,\left(1-\rho^{2}\right)\sigma_{1}^{2}\right)$$X_2$$X_1$

import numpy as np
import numpy.random as nprd

## 
mu1=2
mu2=1
sigma1=np.sqrt(1)
sigma2=np.sqrt(2)
rho=0.5
N=1000 #

## X2
X2=(nprd.randn(N)+mu2)*sigma2
## epsilon
epsilon=nprd.randn(N)*(sigma1*np.sqrt(1-rho**2))
## X1
X1=mu1-rho*sigma1/sigma2*mu2+rho*sigma1/sigma2*X2+epsilon

## 
print("X1,X2\n",np.cov(X1,X2))
print("X1",np.var(X1))
print("X2",np.var(X2))
print("X1,X2",np.corrcoef(X1,X2)[0,1])

## 
import matplotlib.pyplot as plt
## jupyter
%matplotlib inline
# 
plt.rcParams['figure.figsize'] = (10.0, 8.0)
plt.scatter(X2,X1,color='pink') ##
## 
lp=np.linspace(-4,6,20)
y=mu1-rho*sigma1/sigma2*mu2+rho*sigma1/sigma2*lp
plt.plot(lp,y,color='blue') 
plt.xlabel('X2')
plt.ylabel("X1")
plt.title('Relationship of x and y')
plt.show() ##

import numpy as np
import numpy.random as nprd

## 
mu1=2
mu2=1
sigma1=np.sqrt(1)
sigma2=np.sqrt(2)
rho=0.5
N=1000 #
Sigma=np.array([[sigma1**2,rho*sigma1*sigma2],[rho*sigma1*sigma2,sigma2**2]])

## Sigma^0.5
l,L=np.linalg.eig(Sigma)
Lamb=np.array([ll**0.5 for ll in l])
Sigma1p2=np.dot(np.dot(L,np.diag(Lamb)),L.transpose())

## u
U1=nprd.randn(N)
U2=nprd.randn(N)
U=np.array([U1,U2])

## X
X=np.dot(Sigma1p2,U)
X1=X[0,]+mu1
X2=X[1,]+mu2

## 
print("X1,X2\n",np.cov(X1,X2))
print("X1",np.var(X1))
print("X2",np.var(X2))
print("X1,X2",np.corrcoef(X1,X2)[0,1])

## 
import matplotlib.pyplot as plt
## jupyter
%matplotlib inline
# 
plt.rcParams['figure.figsize'] = (10.0, 8.0)
plt.scatter(X2,X1,color='pink') ##
##
